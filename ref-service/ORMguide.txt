How to use
Import any model from the index:

const { Student, Teacher, Subject, ... } = require('../models');
Examples:

Find one: await Student.findOne({ uid: 123 });
List: await Subject.findAll({ where: { id_level: 1 }, orderBy: 'name_subject ASC', limit: 20 });
By ID: await Teacher.findById(5);
Create: await Groupe.create({ code_groupe: 'G1', level_id: 2 });
Update: await Departement.update(3, { name: 'New Name' });
Delete: await Classroom.deleteById(10);
